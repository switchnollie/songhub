factory Message.fromMap(Map<String, dynamic> data, String documentId) {
  if (data == null) {
    return null;
  }
  return Message(
    id: documentId,
    creator: data['creator'],
    content: data['content'],
    createdAt: data['createdAt'],
  );
}
