Future<bool> showDeleteAlert(BuildContext context) async {
  bool result = await showDialog(
    context: context,
    barrierDismissible: false,
    builder: (BuildContext context) {
      return AlertWidget(
        title: 'Delete project',
        text:
            'Are you sure you want to delete this project? This action can\'t be undone and will remove all files and messages too!',
        option1: 'CANCEL',
        option2: 'DELETE',
        onTap: () => _handleSongDelete(context, song),
      );
    },
  );
  if (result != null && result) {
    showSnackBarByContext(context, "Song successfully deleted");
  }
  return result;
}
