Serializes the song to update or add in Firestore

Implementation

Map<String, dynamic> toMap() {
  return {
    'title': title,
    'artist': artist,
    'coverImg': coverImg,
    'participants': participants,
    'lyrics': lyrics,
    'mood': mood,
    'status': status,
    'genre': genre.value,
    'ownedBy': ownedBy,
    'updatedAt': updatedAt,
    'createdAt': createdAt,
  };
}
